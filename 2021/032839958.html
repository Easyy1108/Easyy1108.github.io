排序算法合集

C# 排序算法
发布日期:   2021-03-28

插入排序

插入排序的平均时间复杂度也是 O(n^2)，空间复杂度为常数阶 O(1)，具体时间复杂度和数组的有序性也是有关联的。

插入排序中，当待排序数组是有序时，是最优的情况，只需当前数跟前一个数比较一下就可以了，这时一共需要比较 N-1 次，时间复杂度为 O(N)。最坏的情况是待排序数组是逆序的，此时需要比较次数最多，最坏的情况是 O(n^2)。 

namespace Sort
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] arr = new int[] { 1, 8, 7, 6, 7, 3, 5 };
            for(int i= 0; i < arr.Length; i++)
            {
                for (int j = i;j>0 ; j--)
                {
                    if (arr[j-1] > arr[j])
                    {
                        int temp = arr[j];
                        arr[j] = arr[j-1];
                        arr[j-1] = temp;
                    }
                }
            }
            foreach (var item in arr)
            {
                Console.WriteLine(item);
            }
        }
    }
}

结果： 1356778

选择排序

首先在未排序序列中找到最小（大）元素，存放到排序序列的起始位置。

再从剩余未排序元素中继续寻找最小（大）元素，然后放到已排序序列的末尾。

重复第二步，直到所有元素均排序完毕。 

namespace Sort
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] arr = new int[] { 1, 8, 7, 6, 7, 3, 5 };
            int minIndex, temp;
            for(int i= 0; i < arr.Length-1; i++)
            {
                minIndex = i;
                for(int j =i+1;j<arr.Length; j++)
                {
                    if (arr[j] < arr[minIndex])
                    {
                        minIndex = j;
                    }
                }
                temp = arr[i];
                arr[i] = arr[minIndex];
                arr[minIndex] = temp;
            }
            foreach (var item in arr)
            {
                Console.WriteLine(item);
            }
        }
    }
}

冒泡排序

两个数比较大小，较大的数下沉，较小的数冒起来。 

namespace Sort
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] arr = new int[] { 1, 8, 7, 6, 7, 3, 5 };
            int temp;
            for(int i= 0; i < arr.Length; i++)
            {
                for(int j =0;j<arr.Length-1; j++)
                {
                    if (arr[j] > arr[j+1])
                    {
                        temp = arr[j];
                        arr[j] = arr[j + 1];
                        arr[j + 1] = temp;
                    }
                }
            }            
            foreach (var item in arr)
            {
                Console.WriteLine(item);
            }
        }
    }
}

堆排序

堆排序（Heapsort）是指利用堆这种数据结构所设计的一种排序算法。堆积是一个近似完全二叉树的结构，并同时满足堆积的性质：即子结点的键值或索引总是小于（或者大于）它的父节点。

namespace Sort
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] arr = new int[] { 1, 8, 7, 6, 7, 3, 5 };
            HeapSort(arr);
            foreach (var item in arr)
            {
                Console.WriteLine(item);
            }
        }

        static void HeapSort(int[] arr)
        {
            int len = arr.Length;

            //创建新数组
            int[] tempKey=new int[len + 1];
            for (int i = 0; i < len; i++)
            {
                tempKey[i + 1] = arr[i];
            }

            //初始数据建堆（从最后一个节点的子树开始构建，依次向前，形成二叉堆）
            for (int i = len/2; i >=1; i--)
            {
                Restore(tempKey, i, len);
            }

            //不断输出堆顶元素 并重构堆，排序
            for (int i = len; i>1 ; i--)
            {
                int temp = tempKey[i];
                tempKey[i] = tempKey[1];
                tempKey[1] = temp;
                Restore(tempKey, 1, i - 1);
            }

            //结果返回到arr
            for(int i = 0; i < len; i++)
            {
                arr[i] = tempKey[i + 1];
            }
        }

        //针对已经构建好的二叉堆首尾互换之后的重构
        static void Restore(int[] arr, int rootNode, int nodeCount)
        {
            while (rootNode <= nodeCount / 2)
            {
                int m = (2 * rootNode + 1 <= nodeCount && arr[2 * rootNode + 1] > arr[2 * rootNode]) ? 
                         2 * rootNode + 1:2 * rootNode;
                if (arr[m]>arr[rootNode])
                {
                    int temp = arr[m];
                    arr[m] = arr[rootNode];
                    arr[rootNode] = temp;
                    rootNode = m;
                }
                else
                {
                    break;
                }
            }
        }
    }
}

另一种写法，可能更好理解： 

namespace Sort
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] arr = new int[] { 4, 10, 3, 5, 1, 2 };
            HeapSort(arr, arr.Length);
            foreach (var item in arr)
            {
                Console.WriteLine(item);
            }
        }

        /// <summary>
        /// 堆排序
        /// </summary>
        /// <param name="arr"></param>
        /// <param name="len"></param>
        static void HeapSort(int[] arr, int len)
        {
            //建立最大堆
            BuildMaxHeap(arr, len);

            for (int i = len-1; i>=0 ; i--)
            {
                Swap(arr, i, 0);
                Heapify(arr, i, 0);
            }
        }




        /// <summary>
        /// 建立最大堆，从最后一个父节点，反着遍历，确保是最大堆
        /// </summary>
        /// <param name="arr"></param>
        /// <param name="len"></param>
        static void BuildMaxHeap(int[] arr ,int len)
        {
            int lastNode = len - 1;//最后的节点
            int parentNode = (lastNode - 1 ) / 2;//父节点

            for (int i = parentNode; i>=0; i--)
            {
                Heapify(arr, len, i);
            }
        }


        /// <summary>
        /// 建堆
        /// </summary>
        /// <param name="arr"></param> 数组
        /// <param name="len"></param> 数组长度 相当于节点数
        /// <param name="i"></param>   相当于根节点
        static void Heapify(int[] arr, int len, int i)
        {
            //根节点比节点总数大，直接返回
            if (i >= len)
            {
                return;
            }

            int leftNodeIndex = i * 2 + 1; //左节点
            int rigthNodeIndex = i * 2 + 2;//右节点
            int max = i;              //把当前根节点作为最大值得节点

            //判断左右子节点是不是比根节点大
            if(leftNodeIndex<len&& arr[leftNodeIndex] > arr[max])
            {
                max = leftNodeIndex;
            }
            if(rigthNodeIndex<len&& arr[rigthNodeIndex] > arr[max])
            {
                max = rigthNodeIndex;
            }

            //判断是否需要交换
            if (max != i)
            {
                Swap(arr, i, max);
                Heapify(arr, len, max);
            }
        }



        /// <summary>
        /// 交换
        /// </summary>
        /// <param name="arr"></param>
        /// <param name="i"></param>
        /// <param name="max"></param>
        static void Swap(int[] arr, int i, int max)
        {
            int temp = arr[i];
            arr[i] = arr[max];
            arr[max] = temp;
        }
    }
}

归并排序

归并排序是用分治思想，分治模式在每一层递归上有三个步骤：
分解（Divide）：将n个元素分成个含n/2个元素的子序列。
解决（Conquer）：用合并排序法对两个子序列递归的排序。
合并（Combine）：合并两个已排序的子序列已得到排序结果。
